    /**
     * 可根据时间段筛选查询
     * @param startDate
     * @param endDate
     * @return
     */
    @Override
    public Map<String, Object> downloadAuditList(String startDate, String endDate) {
        Map<String, Object> resultMap = new HashMap<>();
        List<Map<String, Object>> list = new ArrayList<Map<String, Object>>();
        StringBuffer sb = new StringBuffer("select a.auditNo as auditNo, c.name as channelName, b.name as businessPersonName, m.auditMatter as auditMatter, a.auditReason, a.imgUrl, to_char(a.CREATETIME,'yyyy-mm-dd hh24:mi:ss') as createTime from DG_AUDIT a left join  CHANNEL c on c.channelId = a.channelId left join BUSINESSPERSON b on  b.code =a.businessPersonCode left join DG_AUDITMATTER m on  m.matterId= a.auditMatterId where m.isDeleted=" + NORMAL_CODE + " and a.isDeleted=" + NORMAL_CODE);
        // 开始日期
        if (StrKit.notBlank(startDate)) {
            sb.append(" and to_char(a.createTime,'yyyymmdd') >= " + startDate);
        }
        // 结束日期
        if (StrKit.notBlank(endDate)) {
            sb.append(" and to_char(a.createTime,'yyyymmdd') <= " + endDate);
        }
        sb.append("order by  createTime desc");
        try {
            List<Record> listData = Db.find(sb.toString());
            if (listData != null && listData.size() > 0) {
                for (Record rd : listData) {
                    Map<String, Object> map = new HashMap<>();
                    map.put("auditNo", rd.get("auditNo", ""));   //单号
                    map.put("channelName", rd.get("channelName", ""));    // 报备店铺
                    map.put("businessPersonName", rd.get("businessPersonName", ""));   //报备人
                    map.put("auditMatter", rd.get("auditMatter", ""));
                    map.put("auditReason", rd.get("auditReason", ""));
                    map.put("createTime", rd.get("createTime", ""));
                    String url = rd.get("imgUrl", "");
                    String[] img_url = url.split(",");
                    map.put("imgUrl", img_url);
                    list.add(map);
                }
            }
            resultMap.put("dataList", list);// 具体数据
        } catch (Exception e) {
            log.error("获取报备详情列表报错，错误信息：" + e.getMessage());
            return resultMap;
        }
        return resultMap;
    }

# Breadcrumb 面包屑

<template>
  <el-breadcrumb class="app-breadcrumb" separator="/">
    <transition-group name="breadcrumb">
      <el-breadcrumb-item v-for="(item,index) in levelList" v-if="item.meta.title" :key="item.path">
        <span v-if="item.redirect==='noredirect'||index==levelList.length-1" class="no-redirect">{{ item.meta.title }}</span>
        <router-link v-else :to="item.redirect||item.path">{{ item.meta.title }}</router-link>
      </el-breadcrumb-item>
    </transition-group>
  </el-breadcrumb>
</template>

<script>
export default {
  data() {
    return {
      levelList: null
    }
  },
  watch: {
    $route() {
      this.getBreadcrumb()
    }
  },
  created() {
    this.getBreadcrumb()
  },
  methods: {
    getBreadcrumb() {
      let matched = this.$route.matched.filter(item => item.name)
      const first = matched[0]
      if (first && first.name.trim().toLocaleLowerCase() !== 'Dashboard'.toLocaleLowerCase()) {
        matched = [{ path: '/dashboard', meta: { title: '首页' }}].concat(matched)
      }
      this.levelList = matched
    }
  }
}
</script>
