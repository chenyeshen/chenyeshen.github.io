vue+weapp开发小程序 下拉刷新 上拉加载更多 实现

喵ฅ仙 | BLOG 2021-05-01 Vue小程序

# 配置.json文件

在app.json页面配置"enablePullDownRefresh": true，这会让下拉刷新效果适用于项目的每一个页面；或者在对应需要下拉刷新功能的.json文件加上这句配置。

{
  "navigationBarTitleText": "稽核列表",
  "enablePullDownRefresh": true,
  "onReachBottomDistance": 150
}

# 下拉刷新 上拉加载更多

<template>
  <div class="wrap">
    <van-search class="search-content" :value="keyword" @change="onChange" placeholder="搜索报备事项"/>
    <div class="main">
      <!--<div class="channel-title">
        <span>{{channelName}}</span>
      </div>-->
      <div v-for="(item, index) in auditList" :key="index">
        <div>
          <div class="audit-list" @click="goAuditDetail(item.auditId)">
            <div class="title">
              <span>{{item.auditMatter}}</span>
              <span class="icon">></span>
            </div>
            <b class="time">单号：{{item.auditNo}}</b>
            <span class="time">{{item.createtime}}</span>
          </div>
        </div>
      </div>
    </div>
    <div class="footer">
      <van-button style=" position: fixed; bottom: 0;width: 100%;" @click="goAuditAdd" type="info" size="large">新增报备</van-button>
    </div>
  </div>
</template>

<script>
  import {router} from '@/config'
  import {getAuditList} from '@/api/audit'
  import {mapState} from 'vuex'
  import {showLoading} from '../../../api/member'
  import { uploadBuriedPoint, uploadClickBuriedPoint } from '@/api/login'

  export default {
    data() {
      return {
        pageNum: 1,
        pageSize: 10,
        keyword: '',
        auditList: [],
        newAuditList: [],
        channelName: '',
        totalPage: 0
      }
    },
    computed: {
      ...mapState(['guideGuid'])
    },
    onLoad(options) {

    },
    onShow() {
      this.pageNum = 1
      this.auditList = []
      this.getList()
    },
    onPullDownRefresh() {
      this.auditList = []
      this.pageNum = 1
      showLoading()
      setTimeout(() => {
        this.getList()
      }, 1000)
    },
    onReachBottom() {
      if (this.pageNum < this.totalPage) {
        // this.hasMore = true
        this.pageNum += 1
        this.getList()
      }
    },
    mounted() {
      uploadBuriedPoint()
    },

    methods: {
      getList() {
        getAuditList({
          keyword: this.keyword,
          pageNum: this.pageNum,
          pageSize: this.pageSize
        }).then(e => {
          if (e.code === 0) {
            this.auditList = [
              ...this.auditList,
              ...e.msg.dataList
            ]
            /*this.channelName = this.auditList[0].name*/
            this.totalPage = e.msg.totalPage
            // 判断渲染完成
            wx.nextTick(() => {
              wx.hideLoading()
            })
            wx.stopPullDownRefresh()
          }
        }).catch(e => {
          console.log(e)
        })
      },
      // 搜索
      onChange(e) {
        uploadClickBuriedPoint('0058')
        this.keyword = e.mp.detail
        getAuditList({guideGuid: this.guideGuid, keyword: this.keyword}).then(e => {
          if (e.code === 0) {
            this.auditList = e.msg.dataList
          }
        })
      },

      // 稽核详情
      goAuditDetail(e) {
        uploadClickBuriedPoint('0059')
        wx.navigateTo({
          url: router.audit_detail + '?auditId=' + e
        })
      },

      // 稽核列表
      goAuditAdd() {
        uploadClickBuriedPoint('0060')
        wx.navigateTo({
          url: router.audit_add
        })
      },
    }
  }
</script>
