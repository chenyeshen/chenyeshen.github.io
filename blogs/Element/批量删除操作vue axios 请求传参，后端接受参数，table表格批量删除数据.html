           //删除或者批量删除

 /**
     * 删除或者批量删除
     */
    @PostMapping("/delete")
    public Result delete(Integer [] ids) {
        Result result = new Result();
        if (ids ==null){
            return result;
        }
        boolean flag = applicationService.removeByIds(Arrays.asList(ids));
        if (flag) {
            result.setOK();
            return result;
        }
        result.setError();
        return result;
    }

<script>
    import {application_list, getApplicationInfoById,saveOrUpdateApplication} from "../../api/application";
    import {getCompanyNameList, getCompanyInfoById} from "../../api/company";
    import {checkPermission} from "@/utils/permission";
    import qs from 'qs';

    export default {
        data() {
            return {
                applicationId: 0,
                applicationName: '',
                companyId: '',
                remarks: '',
                applicattionType: '',
                applicationVersion: '',
                companyName: '',
                personInCharge: '',
                dialogMode: 1,
                status_options: [
                    {
                        value: 0,
                        label: '可用'
                    }, {
                        value: 1,
                        label: '禁用'
                    },
                ],
                company_options: [],
                companyIdValue: '',
                typeValue: '',
                value: '选择状态',
                show: false,
                search_application_name: '',
                dataList: [],
                keywords: 'keywords',
                pageNum: 1,
                pageSize: 10,
                totalPage: 0,
                dataListLoading: false,
                dataListSelections: [],
                dialogVisible: false,
                ids: []
            }
        },
        created() {
            this.getDataList()
        },

        methods: {
            checkPermission,
            onSearchStateClear(){
                if(this.value ==''){
                    this.value = '选择状态'
                    this.getDataList()
                }
            },

            onSearchStateChange(event){
                this.value = event;
                this.getDataList()
            },

            onSearchNameChange(){
             this.getDataList()
            },
            // 改变应用状态
            stateChange(row){
                let data ={
                    applicationId: row.applicationId,
                    applicationState: row.applicationState
                }
                this.$http.post('/api/interfaceapplication/saveOrUpdateApplication', data).then(res => {
                   if (res.data.success){
                       if (row.applicationState ==1 ){
                           this.$message({
                               message: '禁用',
                               type: 'error',
                               duration: 1000,
                               onClose: () => {
                                   this.getDataList()
                               }
                           })
                       }else {
                           this.$message({
                               message: '可用',
                               type: 'success',
                               duration: 1000,
                               onClose: () => {
                                   this.getDataList()
                               }
                           })
                       }

                   }
                })
            },

            getApplication(id){
                this.applicationId = id || 0
                let data = {
                    applicationId: this.applicationId
                }
                // 获取应用信息
                getApplicationInfoById(data).then(resp => {
                    if (resp.success){
                        this.applicationName = resp.data.applicationName
                        this.companyId = resp.data.companyId
                        this.remarks = resp.data.remarks
                        this.typeValue = resp.data.applicationType
                        this.applicationVersion = resp.data.applicationVersion
                        this.personInCharge = resp.data.personInCharge
                        // 获取公司名称列表
                        getCompanyNameList().then(resp => {
                            if (resp.success){
                                this.company_options = resp.data
                            }
                        })
                        // 获取当前公司名称
                        getCompanyInfoById({
                            companyId: this.companyId
                        }).then(resp => {
                            if (resp.success){
                                this.companyIdValue = resp.data.companyId
                            }
                        })
                    }
                })
            },

            // dialog 公司名称 选择改变
            onCompanyChange(event){
                this.companyId = event
            },

            // 更新或者保存
            saveOrUpdate () {
                   if (this.applicationName ==''){
                       this.$message({
                           message: '应用名为空',
                           type: 'error',
                           duration: 1000,
                           onClose: () => {
                           }
                       })
                      return
                   }

                    let updateData ={
                        applicationId: this.applicationId,
                        applicationName: this.applicationName,
                        companyId: this.companyId,
                        remarks: this.remarks,
                        applicationType: this.typeValue,
                        applicationVersion: this.applicationVersion,
                        personInCharge: this.personInCharge
                    }

                    let insertData ={
                        applicationName: this.applicationName,
                        companyId: this.companyId,
                        remarks: this.remarks,
                        applicationType: this.typeValue,
                        applicationVersion: this.applicationVersion,
                        personInCharge: this.personInCharge
                    }

                  this.$http.post('/api/interfaceapplication/saveOrUpdateApplication', this.dialogMode == 1 ? updateData : insertData).then(resp => {
                    if (resp.data.success){
                        this.dialogVisible = false
                        this.$message({
                            message: '操作成功',
                            type: 'success',
                            duration: 1000,
                            onClose: () => {
                                this.getDataList()
                                this.applicationId = ''
                                this.applicationName = ''
                                this.companyId = ''
                                this.remarks = ''
                                this.typeValue = ''
                                this.companyIdValue = ''
                                this.applicationVersion = ''
                                this.personInCharge = ''
                            }
                        })
                    }else {
                        this.$message.error(resp.data.msg)
                    }
                })
            },

            // 获取数据列表
            getDataList() {
                this.dataListLoading = true
                let data = {
                    pageNum: this.pageNum,
                    pageSize: this.pageSize,
                    keywords: this.search_application_name =='' ? this.keywords : this.search_application_name,
                    applicationState: this.value
                }
                application_list(data).then(resp => {
                    console.info("5555555555")
                    if (resp.success) {
                        this.dataList = resp.data.list
                        this.totalPage = resp.data.totalCount
                    } else {
                        this.dataList = []
                        this.totalPage = 0
                    }
                    this.dataListLoading = false
                })

            },

            // 每页数
            sizeChangeHandle(val) {
                this.pageSize = val
                this.pageNum = 1
                this.getDataList()
            },
            // 当前页
            currentChangeHandle(val) {
                this.pageNum = val
                this.getDataList()
            },
            // 多选
            selectionChangeHandle(val) {
                this.dataListSelections = val
            },
            // 新增 / 修改
            addOrUpdateHandle(item,id) {
                this.applicationName = ''
                this.companyId = ''
                this.remarks = ''
                this.typeValue = ''
                this.applicationVersion = ''
                this.personInCharge = ''

                this.dialogVisible = true
                this.dialogMode = item ? 1 : 0
                //编辑
                if (item){
                    this.companyId = ''
                    this.getApplication(id)
                }else{
                    this.companyIdValue = ''
                    getCompanyNameList().then(resp => {
                        if (resp.success){
                            this.company_options = resp.data
                        }
                    })
                }

            },
            // 删除
             deleteHandle(id) {
                const length = this.dataListSelections.length;
                for (let i = 0; i < length; i++) {     //将每条数据中的id存到ids数组中
                    this.ids.push(this.dataListSelections[i].applicationId);
                }
                var idss = id ? id: this.ids
                this.$confirm(`确定对[id=${id ? id: this.ids}]进行[${id ? '删除' : '批量删除'}]操作?`, '提示', {
                    confirmButtonText: '确定',
                    cancelButtonText: '取消',
                    type: 'warning'
                }).then(() => {
                    this.$http.post(`/api/interfaceapplication/delete?ids=` + idss).then(resp =>{
                        if (resp.data.success){
                            this.$message({
                                message: '操作成功',
                                type: 'success',
                                duration: 1000,
                                onClose: () => {
                                    this.getDataList()
                                    this.ids = []
                                }
                            })
                        } else {
                            this.$message.error(data.msg)
                        }

                    })
                })
            }
        }
    }
</script>
