# serverImp

@Override
public Result getMenuTree(int roleId) {
   Result result = new Result();
   List<SystemMenuEntity> menusBase = systemMenuDao.selectByPid(0,roleId);
   List<SystemMenuEntity> menuLNotBase = systemMenuDao.selectAllNotBase(roleId);
   for (SystemMenuEntity menu : menusBase) {
      List<SystemMenuEntity> menus = iterateMenus(menuLNotBase, menu.getSystemMenuId());
      menu.setChildren(menus);
   }
   result.setOK(menusBase);
   return  result;
}

/**
 *多级菜单查询方法
 * @param menuVoList 不包含最高层次菜单的菜单集合
 * @param pid 父类id
 * @return
 */
public List<SystemMenuEntity> iterateMenus(List<SystemMenuEntity> menuVoList,Integer pid){
   List<SystemMenuEntity> result = new ArrayList<SystemMenuEntity>();
   for (SystemMenuEntity menu : menuVoList) {
      //获取菜单的id
      Integer menuid = menu.getSystemMenuId();
      //获取菜单的父id
      Integer parentid = menu.getSystemMenuParentId();
      if(StringUtils.isNotBlank(parentid.toString())){
         if(parentid.equals(pid)){
            //递归查询当前子菜单的子菜单
            List<SystemMenuEntity> iterateMenu = iterateMenus(menuVoList,menuid);
            menu.setChildren(iterateMenu);
            result.add(menu);
         }
      }
   }
   return result;
}
